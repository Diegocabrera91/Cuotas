/** * PORTAL DE APODERADOS — CUOTAS DEL CURSO (v3) * Backend Google Apps Script (patrón GET / CORS-safe, JSON via ContentService) * * Cambios v3: * - El apoderado ESCRIBE el nombre de su hijo(a) al registrarse; el servidor * valida contra la lista oficial (tolerante a mayúsculas, tildes y espacios) * y guarda el nombre canónico de la hoja. * - Se eliminó la acción pública 'alumnos' por privacidad: la lista de niños * del curso ya no es consultable desde la API. * * DESPLIEGUE: * 1. Pega el ID del Sheets en SPREADSHEET_ID. * 2. En configurarAdmin(), escribe la clave de administrador y EJECÚTALA UNA VEZ * desde el editor (Ejecutar > configurarAdmin). Luego vuelve a dejar el texto * en 'CAMBIA_ESTA_CLAVE'. * 3. Implementar > Administrar implementaciones > lápiz > Versión: NUEVA VERSIÓN * (así la misma URL /exec toma este código). */ const SPREADSHEET_ID = 'PEGA_AQUI_EL_ID_DE_TU_SHEETS'; const HOJA_ALUMNOS = 'Alumnos'; const HOJA_CUOTAS = 'Cuotas'; const HOJA_GASTOS = 'Gastos'; const HOJA_RESUMEN = 'Resumen'; const HOJA_USUARIOS = 'Usuarios'; const MESES_ORDEN = ['marzo','abril','mayo','junio','julio','agosto', 'septiembre','octubre','noviembre','diciembre']; /** * ►► EJECUTAR UNA VEZ desde el editor para fijar la clave de admin. */ function configurarAdmin() { const clave = 'CAMBIA_ESTA_CLAVE'; if (clave === 'CAMBIA_ESTA_CLAVE') { throw new Error('Escribe tu clave real antes de ejecutar configurarAdmin()'); } const hash = sha256Hex_('admin|' + clave); PropertiesService.getScriptProperties().setProperty('ADMIN_HASH', hash); Logger.log('Clave de administrador configurada correctamente.'); } // ───────────────────────────────────────────────────────────── // ROUTER // ───────────────────────────────────────────────────────────── function doGet(e) { const p = (e && e.parameter) || {}; let out; try { switch (p.action) { case 'registro': out = accionRegistro(p); break; case 'login': out = accionLogin(p); break; case 'datos': out = accionDatos(p); break; case 'adminLogin': out = accionAdminLogin(p); break; case 'adminDatos': out = accionAdminDatos(p); break; case 'adminEliminar': out = accionAdminEliminar(p); break; default: out = { ok: false, error: 'Acción no válida' }; } } catch (err) { out = { ok: false, error: 'Error del servidor: ' + String(err && err.message || err) }; } return ContentService .createTextOutput(JSON.stringify(out)) .setMimeType(ContentService.MimeType.JSON); } // ───────────────────────────────────────────────────────────── // UTILIDADES // ───────────────────────────────────────────────────────────── function ss_() { return SpreadsheetApp.openById(SPREADSHEET_ID); } function sha256Hex_(texto) { return Utilities.computeDigest(Utilities.DigestAlgorithm.SHA_256, texto, Utilities.Charset.UTF_8) .map(function (b) { return ((b + 256) % 256).toString(16).padStart(2, '0'); }) .join(''); } /** minúsculas, sin tildes, espacios colapsados — igual que en el frontend */ function norm_(s) { return String(s == null ? '' : s) .trim().toLowerCase() .normalize('NFD').replace(/[\u0300-\u036f]/g, '') .replace(/\s+/g, ' '); } function leerTabla_(nombreHoja, columnaAncla) { const sh = ss_().getSheetByName(nombreHoja); if (!sh) return null; const values = sh.getDataRange().getValues(); if (!values.length) return null; let headerRow = 0; const ancla = norm_(columnaAncla); for (let r = 0; r < Math.min(5, values.length); r++) { if (values[r].some(function (c) { return norm_(c).indexOf(ancla) !== -1; })) { headerRow = r; break; } } const headers = values[headerRow].map(norm_); const rows = values.slice(headerRow + 1) .filter(function (r) { return r.some(function (c) { return String(c).trim() !== ''; }); }); function idx() { for (let a = 0; a < arguments.length; a++) { const t = norm_(arguments[a]); for (let i = 0; i < headers.length; i++) { if (headers[i].indexOf(t) !== -1) return i; } } return -1; } return { headers: headers, rows: rows, idx: idx, sheet: sh }; } function aNumero_(v) { if (typeof v === 'number') return v; const n = parseFloat(String(v).replace(/\./g, '').replace(',', '.').replace(/[^0-9.\-]/g, '')); return isNaN(n) ? 0 : n; } function aFecha_(v) { if (v instanceof Date) { return Utilities.formatDate(v, Session.getScriptTimeZone(), 'dd/MM/yyyy'); } return String(v || '').trim(); } function esPagada_(estado) { return norm_(estado).indexOf('pagad') === 0; } function ahora_() { return Utilities.formatDate(new Date(), Session.getScriptTimeZone(), 'dd/MM/yyyy HH:mm'); } // ───────────────────────────────────────────────────────────── // ALUMNOS Y USUARIOS // ───────────────────────────────────────────────────────────── function listaNombresAlumnos_() { let nombres = []; const tA = leerTabla_(HOJA_ALUMNOS, 'nombre'); if (tA) { const i = tA.idx('nombre'); if (i !== -1) nombres = tA.rows.map(function (r) { return String(r[i]).trim(); }); } if (!nombres.length) { const tC = leerTabla_(HOJA_CUOTAS, 'nombre alumno'); if (tC) { const i = tC.idx('nombre alumno', 'alumno', 'nombre'); if (i !== -1) nombres = tC.rows.map(function (r) { return String(r[i]).trim(); }); } } const vistos = {}; return nombres.filter(function (n) { const k = norm_(n); if (!n || vistos[k]) return false; vistos[k] = true; return true; }); } /** Busca el nombre canónico en la lista oficial a partir de lo que escribió el apoderado */ function buscarAlumnoOficial_(escrito) { const k = norm_(escrito); if (!k) return null; const lista = listaNombresAlumnos_(); for (let i = 0; i < lista.length; i++) { if (norm_(lista[i]) === k) return lista[i]; } return null; } function hojaUsuarios_() { const ss = ss_(); let sh = ss.getSheetByName(HOJA_USUARIOS); if (!sh) { sh = ss.insertSheet(HOJA_USUARIOS); sh.appendRow(['Fecha Registro', 'Alumno', 'Apoderado', 'Hash']); sh.getRange(1, 1, 1, 4).setFontWeight('bold').setBackground('#14604A').setFontColor('#FFFFFF'); sh.setFrozenRows(1); } return sh; } function buscarUsuario_(alumno) { const sh = hojaUsuarios_(); const data = sh.getDataRange().getValues(); const k = norm_(alumno); for (let r = 1; r < data.length; r++) { if (norm_(data[r][1]) === k) { return { fila: r + 1, alumno: String(data[r][1]), apoderado: String(data[r][2] || ''), hash: String(data[r][3] || '') }; } } return null; } // ───────────────────────────────────────────────────────────── // ACCIONES APODERADO // ───────────────────────────────────────────────────────────── function accionRegistro(p) { const escrito = String(p.alumno || '').trim(); const hash = String(p.hash || '').trim(); const apoderado = String(p.apoderado || '').trim(); if (!escrito || !/^[a-f0-9]{64}$/i.test(hash)) { return { ok: false, error: 'Datos de registro incompletos' }; } const canonico = buscarAlumnoOficial_(escrito); if (!canonico) { return { ok: false, error: 'No encontramos ese nombre en la lista del curso. Escríbelo completo, tal como está inscrito (nombre y apellidos). Si el problema persiste, contacta a la tesorería.' }; } const lock = LockService.getScriptLock(); lock.waitLock(10000); try { if (buscarUsuario_(canonico)) { return { ok: false, error: 'Este alumno ya tiene una cuenta creada. Si olvidaste la contraseña, contacta a la tesorería del curso.' }; } hojaUsuarios_().appendRow([new Date(), canonico, apoderado, hash.toLowerCase()]); } finally { lock.releaseLock(); } return { ok: true, alumno: canonico, mensaje: 'Cuenta creada' }; } function validarCredenciales_(p) { const alumno = String(p.alumno || '').trim(); const hash = String(p.hash || '').trim().toLowerCase(); if (!alumno || !hash) return null; const u = buscarUsuario_(alumno); if (!u || u.hash.toLowerCase() !== hash) return null; return u; } function accionLogin(p) { const u = validarCredenciales_(p); if (!u) return { ok: false, error: 'Nombre o contraseña incorrectos. Revisa que el nombre esté escrito igual que al registrarte.' }; return { ok: true, alumno: u.alumno, apoderado: u.apoderado }; } // ───────────────────────────────────────────────────────────── // CÁLCULO DE FONDOS (compartido) // ───────────────────────────────────────────────────────────── function calcularFondos_() { let gastos = []; let totalGastos = 0; const tG = leerTabla_(HOJA_GASTOS, 'monto'); if (tG) { const gMonto = tG.idx('monto', 'valor', 'total'); const gDet = tG.idx('detalle', 'descrip', 'concepto', 'gasto', 'item', 'ítem'); const gFecha = tG.idx('fecha'); if (gMonto !== -1) { gastos = tG.rows.map(function (r) { const m = aNumero_(r[gMonto]); totalGastos += m; return { fecha: gFecha !== -1 ? aFecha_(r[gFecha]) : '', detalle: gDet !== -1 ? String(r[gDet] || '').trim() : '', monto: m }; }).filter(function (g) { return g.monto !== 0 || g.detalle; }); } } let saldoAnterior = 0; const shR = ss_().getSheetByName(HOJA_RESUMEN); if (shR) { const vals = shR.getDataRange().getValues(); outer: for (let r = 0; r < vals.length; r++) { for (let c = 0; c < vals[r].length; c++) { const t = norm_(vals[r][c]); if (t.indexOf('saldo') !== -1 && t.indexOf('anterior') !== -1) { for (let c2 = c + 1; c2 < vals[r].length; c2++) { const n = aNumero_(vals[r][c2]); if (n !== 0 || String(vals[r][c2]).trim() !== '') { saldoAnterior = n; break; } } break outer; } } } } return { gastos: gastos, totalGastos: totalGastos, saldoAnterior: saldoAnterior }; } // ───────────────────────────────────────────────────────────── // DATOS DEL APODERADO // ───────────────────────────────────────────────────────────── function accionDatos(p) { const u = validarCredenciales_(p); if (!u) return { ok: false, error: 'Sesión no válida. Vuelve a ingresar.' }; const tC = leerTabla_(HOJA_CUOTAS, 'nombre alumno'); if (!tC) return { ok: false, error: 'No se encontró la hoja Cuotas' }; const iNom = tC.idx('nombre alumno', 'alumno'); const iMes = tC.idx('mes'); const iMonto = tC.idx('monto'); const iEstado = tC.idx('estado'); const iMetodo = tC.idx('metodo', 'método'); const iFecha = tC.idx('fecha'); const kAlumno = norm_(u.alumno); const misCuotas = []; let recaudadoCurso = 0; let misPagos = 0; tC.rows.forEach(function (r) { const monto = aNumero_(r[iMonto]); const pagada = esPagada_(r[iEstado]); if (pagada) recaudadoCurso += monto; if (norm_(r[iNom]) === kAlumno) { if (pagada) misPagos += monto; misCuotas.push({ mes: String(r[iMes] || '').trim(), monto: monto, estado: pagada ? 'Pagada' : 'Pendiente', metodo: iMetodo !== -1 ? String(r[iMetodo] || '').trim() : '', fecha: iFecha !== -1 ? aFecha_(r[iFecha]) : '' }); } }); misCuotas.sort(function (a, b) { return MESES_ORDEN.indexOf(norm_(a.mes)) - MESES_ORDEN.indexOf(norm_(b.mes)); }); const f = calcularFondos_(); return { ok: true, alumno: u.alumno, apoderado: u.apoderado, cuotas: misCuotas, misPagos: misPagos, recaudado: recaudadoCurso, totalGastos: f.totalGastos, gastos: f.gastos, saldoAnterior: f.saldoAnterior, saldoActual: f.saldoAnterior + recaudadoCurso - f.totalGastos, actualizado: ahora_() }; } // ───────────────────────────────────────────────────────────── // ADMINISTRADOR // ───────────────────────────────────────────────────────────── function validarAdmin_(p) { const guardado = PropertiesService.getScriptProperties().getProperty('ADMIN_HASH'); const hash = String(p.adminHash || '').trim().toLowerCase(); return !!guardado && !!hash && guardado.toLowerCase() === hash; } function accionAdminLogin(p) { if (!PropertiesService.getScriptProperties().getProperty('ADMIN_HASH')) { return { ok: false, error: 'Falta configurar la clave: ejecuta configurarAdmin() en el editor de Apps Script.' }; } if (!validarAdmin_(p)) return { ok: false, error: 'Clave de administrador incorrecta' }; return { ok: true }; } function accionAdminDatos(p) { if (!validarAdmin_(p)) return { ok: false, error: 'Sesión de administrador no válida' }; const tC = leerTabla_(HOJA_CUOTAS, 'nombre alumno'); if (!tC) return { ok: false, error: 'No se encontró la hoja Cuotas' }; const iNom = tC.idx('nombre alumno', 'alumno'); const iMonto = tC.idx('monto'); const iEstado = tC.idx('estado'); const porAlumno = {}; let recaudado = 0; tC.rows.forEach(function (r) { const nombre = String(r[iNom] || '').trim(); if (!nombre) return; const k = norm_(nombre); if (!porAlumno[k]) porAlumno[k] = { nombre: nombre, pagadas: 0, pendientes: 0, total: 0 }; if (esPagada_(r[iEstado])) { const m = aNumero_(r[iMonto]); porAlumno[k].pagadas++; porAlumno[k].total += m; recaudado += m; } else { porAlumno[k].pendientes++; } }); const data = hojaUsuarios_().getDataRange().getValues(); const cuentas = []; const tieneCuenta = {}; for (let r = 1; r < data.length; r++) { const alumno = String(data[r][1] || '').trim(); if (!alumno) continue; tieneCuenta[norm_(alumno)] = true; cuentas.push({ alumno: alumno, apoderado: String(data[r][2] || ''), fecha: aFecha_(data[r][0]) }); } const estado = Object.keys(porAlumno).map(function (k) { const a = porAlumno[k]; return { nombre: a.nombre, pagadas: a.pagadas, pendientes: a.pendientes, total: a.total, cuenta: !!tieneCuenta[k] }; }).sort(function (a, b) { return a.nombre.localeCompare(b.nombre, 'es'); }); const f = calcularFondos_(); return { ok: true, estado: estado, cuentas: cuentas, recaudado: recaudado, totalGastos: f.totalGastos, saldoAnterior: f.saldoAnterior, saldoActual: f.saldoAnterior + recaudado - f.totalGastos, actualizado: ahora_() }; } function accionAdminEliminar(p) { if (!validarAdmin_(p)) return { ok: false, error: 'Sesión de administrador no válida' }; const alumno = String(p.alumno || '').trim(); if (!alumno) return { ok: false, error: 'Falta el nombre del alumno' }; const lock = LockService.getScriptLock(); lock.waitLock(10000); try { const u = buscarUsuario_(alumno); if (!u) return { ok: false, error: 'No existe una cuenta para ese alumno' }; hojaUsuarios_().deleteRow(u.fila); } finally { lock.releaseLock(); } return { ok: true, mensaje: 'Cuenta eliminada. El apoderado puede registrarse nuevamente.' }; }
